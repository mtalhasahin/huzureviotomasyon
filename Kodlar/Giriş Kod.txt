        public string kullanici;
        void baglantim() 
        {
            string kadi = txtKadi.Text;
            string sifre = txtSifre.Text;
            if (txtKadi.Text == "" && txtSifre.Text == "")
            {
                MessageBox.Show("Kullanýcý Adý veya Þifre Boþ Girilemez...");
            }
            try
            {
                if (kadi != "" && sifre != "") 
                {

                    SqlConnection baglanti = new SqlConnection();
                    baglanti.ConnectionString = " Data Source=.;Initial Catalog=HuzurEviOtomasyonu;Integrated Security=True";
                    SqlCommand sorgu = new SqlCommand("SELECT KullaniciAdi,Sifre FROM KullaniciGiris", baglanti);
                    SqlDataReader okuyucu;
                    bool onay = false;
                    try
                    {
                        baglanti.Open();
                        okuyucu = sorgu.ExecuteReader();
                        while (okuyucu.Read())
                        {
                            if ((kadi == okuyucu["KullaniciAdi"].ToString()) && (sifre == okuyucu["Sifre"].ToString()))
                            {

                                onay = true;
                                break;// veri tabanýyla doðru ise kullanýcý giriþi yap
                            }
                        }
                        baglanti.Close();
                    }
                    catch (Exception ex)
                    {
                        MessageBox.Show(ex.Message);
                    }
                    if (onay == true)
                    {
                        kullanici = kadi;
                        mnu.label2.Text = kullanici;
                        mnu.Show();

                        this.Hide();

                    }
                    else
                    {
                        baglanti.Close();
                        MessageBox.Show("Kullanýcý Adý ve Þifreyi Yanlýþ Girdiniz", "Uyarý", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                        txtKadi.Text = "";
                        txtSifre.Text = "";
                    }
                }
            }
            catch (Exception exp)
            {
                MessageBox.Show(exp.ToString());
               
            }
        }
