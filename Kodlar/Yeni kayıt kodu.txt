        private void Temizle(Control ctl)
        {
            //Temizle butonu oluþturuldu tüm grupbox'ýn içindekiler temizlendi
            foreach (Control c in ctl.Controls)
            {
                if (c is TextBox)
                {
                    ((TextBox)c).Clear();
                }
                if (c.Controls.Count > 0)
                {
                    Temizle(c);


                }

            }
        }

        private void textBox4_TextChanged(object sender, EventArgs e)
        {
            if (textBox4.Text.Trim() == "")
            {//tüm kayýtlarý göster

                dt.Clear();
                SqlDataAdapter da = new SqlDataAdapter("Select *From Yenikayit", con);
                da.Fill(dt);
                dataGridView1.DataSource = dt;
            }
            else
            {
                //Kimlik No'ya göre arama gerçekleþti 'Like' kodu ile de aramada eleme yapýldý
                dt.Clear();
                SqlDataAdapter da = new SqlDataAdapter("Select *From Yenikayit where KimlikNo Like '%"+textBox4.Text+"%'", con);
                da.Fill(dt);
                dataGridView1.DataSource = dt;
            
            }
        }
